u16 get_sonic()
{
send_sonic();
TR1 = 1;
while((RX == 1)&&(TF1 == 0));
TR1 = 0;
if(TF1 == 1)
{
TF1 = 0;
}
else
{
time = TH1 *256 + TL1;
distance = (u16)(0.017 * time);
}
TH1 = 0;
TL1 = 0;
return distance;
}

void Timer0Init(void) //2毫秒@12.000MHz
{
AUXR |= 0x80; //定时器时钟1T模式
TMOD &= 0xF0; //设置定时器模式
TL0 = 0x40; //设置定时初始值
TH0 = 0xA2; //设置定时初始值
TF0 = 0; //清除TF0标志
TR0 = 1; //定时器0开始计时
ET0 = 1;
EA = 1;
}
